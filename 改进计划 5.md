```
- 违反透明度原则

```

## 需求描述与定位

### **产品定位**

**新定位**: **Personal Progress Hub（个人进度中心）**

**核心理念**:
- ✅ 真实透明 - 只显示用户自己的真实数据
- ✅ 自我提升 - 强调与自己的过去对比
- ✅ 隐私优先 - 不强迫社交，100% 本地数据
- ✅ 简洁直接 - 专注于个人碳足迹追踪

### **目标用户**

**美国用户特征**:
- 注重隐私保护
- 喜欢真实透明的数据
- 偏好自我提升而非社交竞争
- 重视环保行动的实际效果

### **功能需求**

#### **Must Have（必须有）**

1. **个人数据概览**
   - 总减排量（Total CO₂ Saved）
   - 总活动数（Total Activities）
   - 当前连续记录（Current Streak）
   - 本周减排量（This Week）

2. **历史趋势对比**
   - 本周 vs 上周对比
   - 月度趋势图表
   - 个人最佳纪录

3. **成就系统**
   - 已解锁成就展示
   - 成就进度追踪
   - 新成就提醒

4. **目标与里程碑**
   - 设定减排目标
   - 追踪里程碑进度
   - 达成庆祝

#### **Nice to Have（可以有）**

1. 数据导出功能
2. 每周总结报告
3. 个性化建议
4. 分享功能（可选）

#### **Won't Have（不要有）**

- ❌ 假用户排行榜
- ❌ 虚假社交元素
- ❌ AI 对手挑战
- ❌ 任何误导性内容

### **核心服务实现**

#### ProgressService.swift

```swift
import Foundation

/// Service for calculating personal progress metrics
@MainActor
final class ProgressService: ObservableObject {
    @Published var currentPerformance: PersonalPerformance?
    @Published var weeklyComparison: PerformanceComparison?
    @Published var monthlyTrend: [DailyTotal] = []
    @Published var personalBest: PersonalBest?
    
    private let repository: ActivityRepositoryProtocol
    
    init(repository: ActivityRepositoryProtocol = CoreDataActivityRepository()) {
        self.repository = repository
    }
    
    // MARK: - Public Methods
    
    /// Load all progress data
    func loadProgress() async throws {
        try await loadCurrentPerformance()
        try await loadWeeklyComparison()
        try await loadMonthlyTrend()
        try await loadPersonalBest()
    }
    
    /// Load current week performance
    func loadCurrentPerformance() async throws {
        let calendar = Calendar.current
        let now = Date()
        let startOfWeek = calendar.startOfWeek(for: now)
        
        let activities = try await repository.fetchActivities(
            from: startOfWeek,
            to: now
        )
        
        let totalCO2 = activities.reduce(0.0) { $0 + $1.co2Emission }
        let daysSoFar = max(1, calendar.dateComponents(
            [.day], from: startOfWeek, to: now
        ).day ?? 1)
        
        currentPerformance = PersonalPerformance(
            period: .weekly,
            totalCO2: totalCO2,
            activityCount: activities.count,
            averagePerDay: totalCO2 / Double(daysSoFar)
        )
    }
    
    /// Load weekly comparison (this week vs last week)
    func loadWeeklyComparison() async throws {
        let calendar = Calendar.current
        let now = Date()
        let startOfWeek = calendar.startOfWeek(for: now)
        let startOfLastWeek = calendar.date(
            byAdding: .weekOfYear, value: -1, to: startOfWeek
        )!
        
        // This week
        let thisWeekActivities = try await repository.fetchActivities(
            from: startOfWeek, to: now
        )
        let thisWeekCO2 = thisWeekActivities.reduce(0.0) { $0 + $1.co2Emission }
        
        // Last week
        let lastWeekActivities = try await repository.fetchActivities(
            from: startOfLastWeek, to: startOfWeek
        )
        let lastWeekCO2 = lastWeekActivities.reduce(0.0) { $0 + $1.co2Emission }
        
        let current = PersonalPerformance(
            period: .weekly,
            totalCO2: thisWeekCO2,
            activityCount: thisWeekActivities.count,
            averagePerDay: thisWeekCO2 / 7
        )
        
        let previous = lastWeekCO2 > 0 ? PersonalPerformance(
            period: .weekly,
            totalCO2: lastWeekCO2,
            activityCount: lastWeekActivities.count,
            averagePerDay: lastWeekCO2 / 7
        ) : nil
        
        weeklyComparison = PerformanceComparison(
            current: current,
            previous: previous,
            best: nil
        )
    }
    
    /// Load monthly trend data (past 4 weeks)
    func loadMonthlyTrend() async throws {
        let calendar = Calendar.current
        let now = Date()
        let fourWeeksAgo = calendar.date(
            byAdding: .weekOfYear, value: -4, to: now
        )!
        
        let activities = try await repository.fetchActivities(
            from: fourWeeksAgo, to: now
        )
        
        // Group by week
        monthlyTrend = Dictionary(grouping: activities) { activity in
            calendar.dateComponents(
                [.yearForWeekOfYear, .weekOfYear], from: activity.date
            )
        }.mapValues { activities in
            activities.reduce(0.0) { $0 + $1.co2Emission }
        }.sorted { $0.key < $1.key }
        .map { DailyTotal(
            date: calendar.date(from: $0.key) ?? now,
            totalCO2: $0.value,
            activityCount: $0.value.count
        )}
    }
    
    /// Load personal best records
    func loadPersonalBest() async throws {
        let calendar = Calendar.current
        let now = Date()
        let twelveWeeksAgo = calendar.date(
            byAdding: .weekOfYear, value: -12, to: now
        )!
        
        let allActivities = try await repository.fetchActivities(
            from: twelveWeeksAgo, to: now
        )
        
        // Calculate weekly totals
        let weeklyTotals = Dictionary(grouping: allActivities) { activity in
            calendar.dateComponents(
                [.yearForWeekOfYear, .weekOfYear], from: activity.date
            )
        }.mapValues { $0.reduce(0.0) { $0 + $1.co2Emission } }
        
        // Find best week (lowest CO2)
        let bestWeekCO2 = weeklyTotals.values.min() ?? 0
        
        // Calculate longest streak
        let streak = try await repository.fetchActivityCount(for: .last7Days)
        
        personalBest = PersonalBest(
            bestWeekCO2: bestWeekCO2,
            mostActivitiesInWeek: weeklyTotals.values.max() ?? 0,
            longestStreak: streak
        )
    }
}

/// Personal best records
struct PersonalBest {
    let bestWeekCO2: Double
    let mostActivitiesInWeek: Double
    let longestStreak: Int
}
```

```swift
import SwiftUI

struct ProgressView: View {
    @StateObject private var viewModel = ProgressViewModel()
    @StateObject private var achievementService = AchievementService()
    
    var body: some View {
        ScrollView {
            VStack(spacing: 20) {

                    // 1. Progress Summary

                        currentStreak: viewModel.currentStreak

                    // 2. Weekly Comparison

                    }
                    
                    // 3. Monthly Trend
                    MonthlyTrendChart(data: viewModel.monthlyTrend)
                    
                    // 4. Personal Best
                    if let personalBest = viewModel.personalBest {
                        PersonalBestCard(personalBest: personalBest)
                    }
                    
                    // 5. Achievements
                    AchievementsSection(
                        achievements: achievementService.achievements
                    )
                    
                    // 6. Goals
                    GoalsSection()
                }
            }
            .padding()
        }
        .navigationTitle("My Progress")
        .task {
            await viewModel.loadProgress()
            await achievementService.checkAchievements()
        }
        .refreshable {
            await viewModel.loadProgress()
        }
    }
}
```

## 风险与缓解

### **风险 1: 数据计算错误**
**影响**: 用户数据不准确，降低信任度  
**缓解**: 
- 编写完整的单元测试
- 手动验证计算结果
- 添加数据校验逻辑

### **风险 2: UI 性能问题**
**影响**: 页面卡顿，用户体验差  
**缓解**:
- 使用 LazyVStack
- 优化图表渲染
- 异步加载数据

### **风险 3: 兼容性问题**
**影响**: 旧数据无法显示  
**缓解**:
- 向后兼容测试
- 数据迁移脚本
- 优雅降级处理
