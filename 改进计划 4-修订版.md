## 🔍 问题发现与修复

### **发现的关键问题**

#### ❌ **问题 1: 数据结构与现有代码不兼容**
**原计划：**
```swift
struct PersonalPerformance: Identifiable, Codable {
    let id: String
    let period: PerformancePeriod
    // ... 太多属性
}
```

**问题：**
- 不需要 `Identifiable`（不是列表项）
- 不需要 `id` 字段（不是数据库实体）
- 过于复杂，不符合现有代码简洁风格

**修复：**
```swift
struct PersonalPerformance {
    let period: PerformancePeriod
    let totalCO2: Double
    let activityCount: Int
    let averagePerDay: Double
    
    var formattedCO2: String {
        String(format: "%.1f lbs", totalCO2)
    }
}
```

---

#### ❌ **问题 2: 使用了不存在的 Repository 方法**
**原计划：**
```swift
func loadPerformanceComparison() async throws {
    // 直接使用不存在的方法
    let activities = try await repository.fetchActivities(...)
}
```

**问题：**
- `ActivityRepositoryProtocol` 已有 `fetchActivities(from:to:)` 方法
- 但不需要重新实现，应该复用现有方法
- 测试用例中使用了不存在的 `MockActivityRepository` 属性

**修复：**
- 使用现有的 `ActivityRepositoryProtocol` 方法
- 删除不必要的 Mock 测试代码，使用真实 Repository 测试

---

#### ❌ **问题 3: AchievementRequirement 枚举不匹配**
**原计划：**
```swift
enum AchievementRequirement: Codable {
    case percentileRanking(Double)  // ❌ 不存在
    case beatPersonalBest           // ❌ 不存在
    case perfectWeek                // ❌ 不存在
}
```

**实际现有代码：**
```swift
enum AchievementRequirement: Codable {
    case totalActivities(count: Int)
    case streakDays(days: Int)
    case totalCO2Saved(kg: Double)
    case weeklyActivities(week: Int, count: Int)
    case categoryMilestone(category: String, count: Int)
    case consecutiveDays(days: Int)
}
```

**修复：**
- 扩展现有枚举而不是重新定义
- 使用已有的 `totalCO2Saved` 和 `streakDays` 实现新成就

---

#### ❌ **问题 4: 使用了不存在的 Fonts 和 AppColors**
**原计划：**
```swift
.font(Fonts.title2)  // ✅ 存在
.font(Fonts.headline)  // ✅ 存在
.font(Fonts.caption1)  // ❌ 不存在（应该是 .caption 或 .footnote）
```

**实际现有代码：**
```swift
struct Fonts {
    static var largeTitle: Font
    static var title1: Font
    static var title2: Font
    static var title3: Font
    static var headline: Font
    static var body: Font
    static var callout: Font
    static var subheadline: Font
    static var footnote: Font  // ← 使用这个而不是 caption1
    static var caption1: Font  // ← 检查是否存在
    static var caption2: Font
}
```

**修复：**
- 使用实际存在的字体名称
- 统一使用 `.footnote` 替代 `.caption1`

#### ❌ **问题 5: 测试用例过于复杂**

```swift
import Foundation

@MainActor
final class PersonalRankingService: ObservableObject {
    @Published var currentWeekPerformance: PersonalPerformance?
    @Published var performanceComparison: PerformanceComparison?
    @Published var percentileRanking: PercentileRanking?
    
    private let repository: ActivityRepositoryProtocol
    
    init(repository: ActivityRepositoryProtocol = CoreDataActivityRepository()) {
        self.repository = repository
    }
    
    // MARK: - Public Methods
    
    /// 加载本周表现数据
    func loadCurrentWeekPerformance() async throws {
        let calendar = Calendar.current
        let now = Date()
        let startOfWeek = calendar.startOfWeek(for: now)
        
        let activities = try await repository.fetchActivities(
            from: startOfWeek,
            to: now
        )
        
        let totalCO2 = activities.reduce(0.0) { $0 + $1.co2Emission }
        let daysSoFar = max(1, calendar.dateComponents([.day], from: startOfWeek, to: now).day ?? 1)
        let averagePerDay = totalCO2 / Double(daysSoFar)
        
        currentWeekPerformance = PersonalPerformance(
            period: .weekly,
            totalCO2: totalCO2,
            activityCount: activities.count,
            averagePerDay: averagePerDay
        )
    }
    
    /// 加载历史对比数据
    func loadPerformanceComparison() async throws {
        let calendar = Calendar.current
        let now = Date()
        let startOfWeek = calendar.startOfWeek(for: now)
        
        // 本周
        let currentWeekActivities = try await repository.fetchActivities(
            from: startOfWeek,
            to: now
        )
        let currentWeekCO2 = currentWeekActivities.reduce(0.0) { $0 + $1.co2Emission }
        
        // 上周
        let lastWeekStart = calendar.date(byAdding: .weekOfYear, value: -1, to: startOfWeek)!
        let lastWeekActivities = try await repository.fetchActivities(
            from: lastWeekStart,
            to: startOfWeek
        )
        let lastWeekCO2 = lastWeekActivities.reduce(0.0) { $0 + $1.co2Emission }
        
        // 历史最佳（过去 12 周）
        let twelveWeeksAgo = calendar.date(byAdding: .weekOfYear, value: -12, to: startOfWeek)!
        let allActivities = try await repository.fetchActivities(
            from: twelveWeeksAgo,
            to: now
        )
        
        let weeklyTotals = Dictionary(grouping: allActivities) { activity in
            calendar.dateComponents([.yearForWeekOfYear, .weekOfYear], from: activity.date)
        }.mapValues { $0.reduce(0.0) { $0 + $1.co2Emission } }
        
        let bestWeekCO2 = weeklyTotals.values.max() ?? 0
        
        // 构建对比对象
        let current = PersonalPerformance(
            period: .weekly,
            totalCO2: currentWeekCO2,
            activityCount: currentWeekActivities.count,
            averagePerDay: currentWeekCO2 / 7
        )
        
        let previous = lastWeekCO2 > 0 ? PersonalPerformance(
            period: .weekly,
            totalCO2: lastWeekCO2,
            activityCount: lastWeekActivities.count,
            averagePerDay: lastWeekCO2 / 7
        ) : nil
        
        let best = bestWeekCO2 > 0 ? PersonalPerformance(
            period: .weekly,
            totalCO2: bestWeekCO2,
            activityCount: 0,
            averagePerDay: bestWeekCO2 / 7
        ) : nil
        
        performanceComparison = PerformanceComparison(
            current: current,
            previous: previous,
            best: best
        )
    }
    
    /// 计算百分位排名
    func calculatePercentileRanking() async throws {
        guard let current = currentWeekPerformance else { return }
        
        let calendar = Calendar.current
        let now = Date()
        let twelveWeeksAgo = calendar.date(byAdding: .weekOfYear, value: -12, to: now)!
        
        let allActivities = try await repository.fetchActivities(
            from: twelveWeeksAgo,
            to: now
        )
        
        let weeklyTotals = Dictionary(grouping: allActivities) { activity in
            calendar.dateComponents([.yearForWeekOfYear, .weekOfYear], from: activity.date)
        }.mapValues { $0.reduce(0.0) { $0 + $1.co2Emission } }
        
        let allWeeks = Array(weeklyTotals.values)
        let weeksWithLessCO2 = allWeeks.filter { $0 < current.totalCO2 }.count
        
        let percentile = allWeeks.isEmpty ? 0 : 
            Double(weeksWithLessCO2) / Double(allWeeks.count) * 100
        
        percentileRanking = PercentileRanking.from(percentile: percentile)
    }
}

// MARK: - Data Models

/// 个人表现数据（简化版）
struct PersonalPerformance {
    let period: PerformancePeriod
    let totalCO2: Double
    let activityCount: Int
    let averagePerDay: Double
    
    var formattedCO2: String {
        String(format: "%.1f lbs", totalCO2)
    }
}

/// 表现对比结果
struct PerformanceComparison {
    let current: PersonalPerformance
    let previous: PersonalPerformance?
    let best: PersonalPerformance?
    
    var changePercent: Double {
        guard let previous = previous, previous.totalCO2 > 0 else { return 0 }
        return ((current.totalCO2 - previous.totalCO2) / previous.totalCO2) * 100
    }
    
    var isImproved: Bool {
        changePercent < 0  // 负数表示减排更多
    }
    
    var changeDescription: String {
        if changePercent == 0 { return "No change" }
        let sign = changePercent < 0 ? "↓" : "↑"
        return "\(sign) \(String(format: "%.1f", abs(changePercent)))%"
    }
}

/// 时间周期
enum PerformancePeriod: String, Codable {
    case weekly = "Week"
    case monthly = "Month"
    case yearly = "Year"
}

/// 百分位排名结果
struct PercentileRanking {
    let percentile: Double
    let rank: String
    let message: String
    let icon: String
    
    static func from(percentile: Double) -> PercentileRanking {
        switch percentile {
        case 90...:
            return PercentileRanking(
                percentile: percentile,
                rank: "Top 10%",
                message: "Best performance ever! You're beating 90% of your history!",
                icon: "trophy.fill"
            )
        case 70..<90:
            return PercentileRanking(
                percentile: percentile,
                rank: "Great",
                message: "Great job! Better than most of your history!",
                icon: "star.fill"
            )
        case 50..<70:
            return PercentileRanking(
                percentile: percentile,
                rank: "Good",
                message: "Good effort! Above average!",
                icon: "checkmark.circle.fill"
            )
        default:
            return PercentileRanking(
                percentile: percentile,
                rank: "Keep Going",
                message: "Keep going! Every step counts!",
                icon: "leaf.fill"
            )
        }
    }
}
```

```swift
import XCTest
@testable import suslife

@MainActor
final class PersonalRankingServiceTests: XCTestCase {
    
    var sut: PersonalRankingService!
    
    override func setUp() async throws {
        sut = PersonalRankingService()
    }
    
    override func tearDown() async throws {
        sut = nil
    }
    
    func testLoadCurrentWeekPerformance_WithNoData() async throws {
        // When: 没有数据
        try await sut.loadCurrentWeekPerformance()
        
        // Then: 应该有 0 个活动
        XCTAssertNotNil(sut.currentWeekPerformance)
        XCTAssertEqual(sut.currentWeekPerformance?.activityCount, 0)
        XCTAssertEqual(sut.currentWeekPerformance?.totalCO2, 0.0, accuracy: 0.01)
    }
    
    func testPerformanceComparison_ChangePercent() async {
        // Given: 创建对比数据
        let current = PersonalPerformance(
            period: .weekly,
            totalCO2: 20.0,
            activityCount: 5,
            averagePerDay: 2.86
        )
        let previous = PersonalPerformance(
            period: .weekly,
            totalCO2: 25.0,
            activityCount: 6,
            averagePerDay: 3.57
        )
        
        let comparison = PerformanceComparison(
            current: current,
            previous: previous,
            best: nil
        )
        
        // Then: 变化百分比应该是 -20%
        XCTAssertEqual(comparison.changePercent, -20.0, accuracy: 0.01)
        XCTAssertTrue(comparison.isImproved)
        XCTAssertEqual(comparison.changeDescription, "↓ 20.0%")
    }
    
    func testPercentileRanking_Top10() {
        // When: 百分位 95
        let ranking = PercentileRanking.from(percentile: 95.0)
        
        // Then: 应该是 Top 10%
        XCTAssertEqual(ranking.rank, "Top 10%")
        XCTAssertEqual(ranking.icon, "trophy.fill")
        XCTAssertTrue(ranking.message.contains("90%"))
    }
    
    func testPercentileRanking_Average() {
        // When: 百分位 45
        let ranking = PercentileRanking.from(percentile: 45.0)
        
        // Then: 应该是 Keep Going
        XCTAssertEqual(ranking.rank, "Keep Going")
        XCTAssertEqual(ranking.icon, "leaf.fill")
    }
}
```

```swift
extension AchievementService {
    /// 检查个人里程碑成就
    func checkPersonalMilestones() async {
        do {
            let totalCO2Saved = try await repository.calculateTotalCO2Saved()
            let totalActivities = try await repository.fetchTotalActivities()
            let streak = try await repository.fetchActivityCount(for: .last7Days)
            
            // 检查所有个人里程碑成就
            for achievement in Achievement.personalMilestones {
                if let index = achievements.firstIndex(where: { $0.id == achievement.id }) {
                    if achievements[index].isUnlocked { continue }
                    
                    let shouldUnlock = checkRequirement(
                        achievements[index].requirement,
                        totalActivities: totalActivities,
                        totalCO2Saved: totalCO2Saved,
                        streakDays: streak
                    )
                    
                    if shouldUnlock {
                        unlockAchievement(at: index)
                        await handleAchievementUnlock(achievements[index])
                    }
                }
            }
        } catch {
            print("Error checking personal milestones: \(error)")
        }
```

```swift
struct PersonalLeaderboardView: View {

    @State private var showingShareSheet = false
    
    var body: some View {
        ScrollView {
            VStack(spacing: 20) {
                // 1. 百分位排名卡片
                percentileRankingCard
                
                // 2. 本周表现概览

        }

        VStack(spacing: 16) {

                Image(systemName: ranking.icon)
                    .font(.system(size: 50))
                    .foregroundColor(.yellow)
                
                Text(ranking.rank)
                    .font(Fonts.title2)
                    .fontWeight(.bold)
                    .foregroundColor(AppColors.textPrimary)
                
                Text(ranking.message)
                    .font(Fonts.body)
                    .foregroundColor(AppColors.textSecondary)
                    .multilineTextAlignment(.center)
                    .padding(.horizontal)

            }
        }
        .padding(32)
        .frame(maxWidth: .infinity)
        .background(AppColors.cardBackground)
        .cornerRadius(16)
    }

        VStack(alignment: .leading, spacing: 16) {
            HStack {
                Image(systemName: "calendar")
                    .foregroundColor(AppColors.primary)
                Text("This Week")
                    .font(Fonts.headline)
                    .foregroundColor(AppColors.textPrimary)
            }
            
            if let performance = rankingService.currentWeekPerformance {
                HStack(spacing: 20) {
                    StatView(
                        title: "Total CO₂",
                        value: performance.formattedCO2,
                        icon: "leaf.fill"
                    )
                    
                    StatView(
                        title: "Activities",
                        value: "\(performance.activityCount)",

                        title: "Avg/Day",
```
